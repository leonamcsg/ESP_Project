138// function that fetches datetime from pool.ntp.org
139static void ntp_fetchData(void)
140{
141 int sockfd;
142 struct sockaddr_in serv_addr;
143 struct hostent *server;
144 ntp_packet packet;
145 int n;
146 time_t txTm;
147 struct tm *local_time;
148
149 memset(&serv_addr, 0, sizeof(serv_addr));
150 memset(&packet, 0, sizeof(ntp_packet));
151
152
153 // Create a UDP socket
154 sockfd = socket(AF_INET, UDP_SOCKET, ip_protocol);
155 if (sockfd < 0) {
156 ESP_LOGE(TAG, "ERROR opening socket");
157 return;
158 }
159
160 // Get server address information
161 server = gethostbyname(NTP_SERVER);
162 if (server == NULL) {
163 ESP_LOGE(TAG, "ERROR, no such host\n");
164 return;
165 }
166
167 serv_addr.sin_family = AF_INET;
168 memcpy(&serv_addr.sin_addr.s_addr, server->h_addr, server->h_length);
169 serv_addr.sin_port = htons(NTP_PORT);
170
171 // Initialize NTP packet
172 packet.li_vn_mode = 0x1B; // li = 0, vn = 3, mode = 3
173
174 // Send NTP request
175 n = sendto(sockfd, &packet, sizeof(ntp_packet), 0, (struct sockaddr *)&serv_addr, sizeof(serv_addr));
176 if (n < 0) {
177 ESP_LOGE(TAG, "ERROR writing to socket");
178 return;
179 }
180
181 // Receive NTP response
182 n = recvfrom(sockfd, &packet, sizeof(ntp_packet), 0, NULL, NULL);
183 if (n < 0) {
184 ESP_LOGE(TAG, "ERROR reading from socket");
185 return;
186 }
187
188 // Parse NTP response
189 packet.txTm_s = ntohl(packet.txTm_s);
190 packet.txTm_f = ntohl(packet.txTm_f);
191
192 // Calculate Unix timestamp
193 txTm = packet.txTm_s - NTP_TIMESTAMP_DELTA;
194
195 // Set timezone (example for UTC)
196 setenv("TZ", "UTC+3", 1);
197 tzset();
198
199 // Convert Unix timestamp to local time
200 local_time = localtime(&txTm);
201
202 // Format date and time separately
203 memset(date_str, 0, DATE_LEN);
204 strftime(date_str, sizeof(date_str), "%d/%m/%Y", local_time);
205 memset(time_str, 0, TIME_LEN);
206 strftime(time_str, sizeof(time_str), "%H:%M", local_time);
207
208 // Print the date and time
209 printf("Date: %s\n", date_str);
210 printf("Time: %s\n", time_str);
211
212 close(sockfd);
213}
