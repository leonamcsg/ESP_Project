93static esp_err_t APP_URI_FUNCTION_HANDLER_NAME(wifi_connect_json)(httpd_req_t *req)
94{
95 // char localJSONObjBuffer[BUFFER_MAX_SIZE];
96 size_t lenBodyJson = 0;
97 size_t lenPass;
98 size_t lenSSID;
99
100 char * ssid_p;
101 char * pwd_p;
102 wifi_config_t * wifi_config_p = NULL;
103
104 ESP_LOGI(TAG, "/wifiConnect.json requested");
105
106 memset(localJSONObjBuffer,0, BUFFER_MAX_SIZE);
107
108 // Get Request Body
109 lenBodyJson = req->content_len;
110 httpd_req_recv(req, localJSONObjBuffer, lenBodyJson);
111 printf("Data sent by the client: %.*s\n",lenBodyJson, localJSONObjBuffer);
112
113 // Parse the JSON data (example using cJSON)
114 cJSON *body_json = cJSON_Parse(localJSONObjBuffer);
115 if (!body_json) {
116 ESP_LOGE(TAG, "Failed to parse JSON data");
117 cJSON_Delete(body_json);
118 return ESP_FAIL;
119 }
120
121 cJSON *ssid_json = cJSON_GetObjectItemCaseSensitive(body_json, "c_ssid");
122 cJSON *pwd_json = cJSON_GetObjectItemCaseSensitive(body_json, "c_pwd");
123
124 if (!ssid_json || !pwd_json) {
125 ESP_LOGE(TAG, "Missing 'c_ssid' or 'c_pwd' in JSON data");
126 cJSON_Delete(body_json);
127 return ESP_FAIL;
128 }
129
130 if (!cJSON_IsString(ssid_json) || !cJSON_IsString(pwd_json)) {
131 ESP_LOGE(TAG, "Invalid data type for 'c_ssid' or 'c_pwd'");
132 cJSON_Delete(body_json);
133 return ESP_FAIL;
134 }
135
136
137 // Get SSID from body
138 ssid_p = wifiApp_getStationSSID();
139 memset(ssid_p, 0x00, WIFI_SSID_LENGTH);
140 ssid_p = ssid_json->valuestring;
141 lenSSID = strlen(ssid_p);
142
143 // Get Password from body
144 pwd_p = wifiApp_getStationPassword();
145 memset(pwd_p, 0x00, WIFI_PASSWORD_LENGTH);
146 pwd_p = pwd_json->valuestring;
147 lenPass = strlen(pwd_p);
148
149 // Update the WiFi networks configuration and let the WiFi applications know
150 wifi_config_p = wifiApp_getWifiConfig();
151 memset(wifi_config_p, 0x00, sizeof(wifi_config_t));
152 memcpy(wifi_config_p->sta.ssid, ssid_p, lenSSID);
153 memcpy(wifi_config_p->sta.password, pwd_p, lenPass);
154 wifiApp_sendMessage(WIFI_APP_CONNECTING_FROM_HTTP_SERVER);
155
156 cJSON_Delete(body_json);
157
158 return ESP_OK;
159}

188static esp_err_t APP_URI_FUNCTION_HANDLER_NAME(get_wifi_connect_info_json)(httpd_req_t *req)
189{
190 ESP_LOGI(TAG, "/wifiConnectInfo.json requested");
191
192 char ipInfoJSON[200];
193 memset(ipInfoJSON, 0x00, sizeof(ipInfoJSON));
194
195 char ip[IP4ADDR_STRLEN_MAX];
196 char netmask[IP4ADDR_STRLEN_MAX];
197 char gateway[IP4ADDR_STRLEN_MAX];
198
199 if (*httpServer_get_wifiConnectStatus() == HTTP_WIFI_STATUS_CONNECT_SUCCESS)
200 {
201 wifi_ap_record_t wifi_data;
202 ESP_ERROR_CHECK(esp_wifi_sta_get_ap_info(&wifi_data));
203 char * ssid = (char*)wifi_data.ssid;
204
205 esp_netif_ip_info_t ip_info;
206 ESP_ERROR_CHECK(esp_netif_get_ip_info(esp_netif_sta, &ip_info));
207 esp_ip4addr_ntoa(&ip_info.ip, ip, IP4ADDR_STRLEN_MAX);
208 esp_ip4addr_ntoa(&ip_info.netmask, netmask, IP4ADDR_STRLEN_MAX);
209 esp_ip4addr_ntoa(&ip_info.gw, gateway, IP4ADDR_STRLEN_MAX);
210
211 sprintf(ipInfoJSON, "{\"ip\":\"%s\",\"netmask\":\"%s\",\"gateway\":\"%s\",\"ap\":\"%s\"}", ip, netmask, gateway, ssid);
212 }
213
214 httpd_resp_set_type(req, "application/json");
215 httpd_resp_send(req, ipInfoJSON, strlen(ipInfoJSON));
216
217 return ESP_OK;
218}
