162static void httpServer_freeRTOS_setup(void)
163{
164 // Create HTTP server monitor task
165 xTaskCreatePinnedToCore(&httpServer_freeRTOS_monitor,
166 "httpServer_monitor",
167 HTTP_SERVER_MONITOR_STACK_SIZE,
168 NULL,
169 HTTP_SERVER_MONITOR_PRIORITY,
170 &task_http_server_monitor,
171 HTTP_SERVER_MONITOR_CORE_ID);
172
173 // Create the message queue
174 http_server_monitor_queue_handle = xQueueCreate(3, sizeof(http_server_queue_message_t));
175}

177static void httpServer_freeRTOS_endTask(void)
178{
179 if(task_http_server_monitor)
180 {
181 vTaskDelete(task_http_server_monitor);
182 ESP_LOGI(TAG, "httpServer_freeRTOS_endTask: stopping HTTP server monitor");
183 task_http_server_monitor = NULL;
184 }
185}

224static void httpServer_configure(httpd_config_t * config)
225{
226// *config = HTTPD_DEFAULT_CONFIG();
227
228 // The core that the HTTP server will run on
229 config->core_id = HTTP_SERVER_TASK_CORE_ID;
230
231 // Adjust the default priority to 1 less than the WiFi application task
232 config->task_priority = HTTP_SERVER_TASK_PRIORITY;
233
234 // Bump up the stack size (default is 4096)
235 config->stack_size = HTTP_SERVER_STACK_SIZE;
236
237 // Increase uri handlers
238 config->max_uri_handlers = HTTP_SERVER_MAX_URI_HANDLERS;
239
240 // Increase the timeout limits
241 config->recv_wait_timeout = HTTP_SERVER_TIMEOUT_LIMIT;
242 config->send_wait_timeout = HTTP_SERVER_TIMEOUT_LIMIT;
243
244 ESP_LOGI(TAG, "httpServer_configure: Starting server on port: '%d' with task priority: '%d'",
245 config->server_port,
246 config->task_priority);
247}

273void httpServer_start(void)
274{
275 if (http_server_handle == NULL)
276 {
277 // Initializing and configurating the structure
278 httpd_config_t config = HTTPD_DEFAULT_CONFIG();
279 httpServer_configure(&config);
280
281 // Create HTTP server monitor task and the message queue
282 httpServer_freeRTOS_setup();
283
284 //Start the httpd server
285 if (httpd_start(&http_server_handle, &config) == ESP_OK)
286 {
287 ESP_LOGI(TAG, "httpServer_configure: Registering URI handlers");
288 httpServer_uri_setFilesHandlersAndRoutes();
289 httpServer_uri_setRoutesFromOtherFiles();
290 } else
291 {
292 http_server_handle = NULL;
293 }
294 }
295}
